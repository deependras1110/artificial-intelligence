(operator
 moveFromPegToPeg
 (params
  (<fromPeg> Peg) (<toPeg> Peg) (<disk> Disk))
 (preconds
  (on <disk> <fromPeg>) (clear <toPeg>) (clear <disk>))
 (effects
  (clear <fromPeg>) (on <disk> <toPeg>) (del clear <toPeg>) (del on <disk> <fromPeg>)))
